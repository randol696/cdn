Navbar & sidebar

The bar and drawer framing this whole site are rg.ui.navbar() and rg.ui.sidebar() — reusable on any page. Resize the window below 900px to see the sidebar collapse into the hamburger menu.

const navbar = rg.ui.navbar({
  brand: 'My App',
  links: [
    { label: 'Home', href: '/', active: true },
    { label: 'Docs', href: '/docs' }
  ],
  actions: [rg.ui.themeToggle()],
  onToggle: () => sidebar.classList.toggle('rg-sidebar--open')
});

const sidebar = rg.ui.sidebar({
  sections: [
    { title: 'Guide', links: [{ label: 'Getting started', href: '#', active: true }] }
  ]
});

document.body.prepend(sidebar, navbar);
document.body.classList.add('rg-with-navbar', 'rg-with-sidebar');
